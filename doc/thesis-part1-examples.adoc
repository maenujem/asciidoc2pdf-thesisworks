.Java code
[source,java]
----
public class Stack {

    private int size;
    private int element;

    public boolean isEmpty() {
        return size == 0;
    }

    public int pop() {
        if (size == 0) {
            throw new EmptyStackPoppedException();
        }
        size--;
        return element;
    }
}
----
